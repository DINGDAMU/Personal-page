"use strict";

// SEE: http://algo.inria.fr/flajolet/Publications/FlFuGaMe07.pdf
// SEE: http://blog.aggregateknowledge.com/2012/10/25/sketch-of-the-day-hyperloglog-cornerstone-of-a-big-data-infrastructure

// formatting
var integerFormat = d3.format(",.0f");
var numberFormat = d3.format(",f");
var percentFormat = d3.format(".1f");
var siFormat = d3.format(".2s")/*TODO: want 'B' over 'G', etc*/;

// constants
var r = 4/*register width*/;
var maxCardinality = Math.pow(2, Math.pow(2, r)) - 1/*the maximum allowed cardinality (log-log)*/;
var b = 6/*number of bits (from which 'm' is defined) (this is 'k' in the blog post and in "Loglog Counting of Large Cardinalities")*/;
var registerIndexMask = (1 << b) - 1/*the mask applied to the hashed value to get the register index*/;
var valueBits = Math.pow(2, r) - 1/*the max value in a register*/;
var maxRegisterValue = valueBits/*synonym*/;
var m = Math.pow(2, b)/*number of registers*/;
var hllError = 1.04 / Math.sqrt(m)/*theoretical HLL error*/;
var hllErrorPercent = hllError * 100/*%*//*for convenience*/;
var keepHashBits = 24/*at least b + (2^r - 1)*/;
var maxValue = Math.pow(2, keepHashBits) - 1/*the largest allowed value to minimize hash collisions (arbitrarily linked to the hash bits)*/;

var ALGO_BALL_BIN = "Ball-Bin";
var ALGO_HLL = "HLL";

var tableCardinalities = [ { name: "LogLog", className: "ll", cardinality: function(d) { return d.llEstimatedCardinality; }, error: function(d) { return d.llError; } },
{ name: "HyperLogLog", className: "hll", cardinality: function(d) { return d.hllEstimatedCardinality; }, error: function(d) { return d.hllError; } },
/* NOTE: still a work in progress: { name: "Expectation", className: "expectation", cardinality: function(d) { return d.expectationEstimatedCardinality; }, error: function(d) { return d.expectationError; } }*/ ];
var algorithms = [ { name: "LogLog", className: "ll" },
{ name: "Ball-Bin", className: "ball-bin" },
{ name: "HyperLogLog", className: "hll" } ];
var distributions = [ { name: "Observed", className: "actual-data" },
{ name: "Theoretical", className: "theoretical-data" } ]/*must match series order*/;
var means = [ { name: "Arithmetic Mean", className: "arithmetic-mean", accessor: function(d) { return d.arithmeticMean; } },
{ name: "Geometric Mean", className: "geometric-mean", accessor: function(d) { return d.geometricMean; } },
{ name: "Harmonic Mean", className: "harmonic-mean", accessor: function(d) { return d.harmonicMean; } } ];

// state
var valueSet = new Object()/*the set of all values seen (used to get the cardinality)*/;
var registerValues = new Array(m)/*M[j]*/;
blogUtil.fillArray(registerValues, m, 0/*initialize*/);
var registerValueDistribution = new Array(maxRegisterValue + 1/*to include 0*/)/*regenerated from registerValues each step*/;
blogUtil.fillArray(registerValueDistribution, (maxRegisterValue + 1), 0/*initialize*/);
var theoreticalRegisterValueDistribution = new Array(maxRegisterValue + 1/*to include 0*/)/*computed each step*/;
blogUtil.fillArray(theoreticalRegisterValueDistribution, (maxRegisterValue + 1), 0/*initialize*/);
function rho(value/*w*/) {
// NOTE: there are two approaches: mask "value" so that there are 1's
// in the upper bits or do a min. The former is more performant
// the latter is easier to read.
var value = blogUtil.leastSignificantBit(value) + 1/*since 1-based*/;
return Math.min(maxRegisterValue, value);
};
var steps = []/*all stats for all steps*/;
steps.push({
index: 0/*for convenience*/,
cardinality: 0,
value: 0,
hashValue: 0,
registerIndex: 0/*none*/,
registerValue: 0,
llEstimatedCardinality: 0,
llError: 0,
hllEstimatedCardinality: 0,
hllCardinalityAlgorithm: ALGO_BALL_BIN/*so that it matches at 0*/,
hllError: 0,
expectationEstimatedCardinality: 0,
expectationError: 0
})/*0th 'empty' value (for a 'pretty' default)*/;

// UI
var stopped = true;
var delay = 100/*ms*/;
var lastTime = 0;

var series = [ registerValueDistribution/*primary*/,
theoreticalRegisterValueDistribution ];

var startStopButton;
var inputDelay, inputStepSize, inputAutoScale;
var bitVectorGraphic, registerValuesGraphic;
var cardinalityTableGraphic;
var errorChart;
var registerValueHistogram;

var container = d3.select("#hll");

createControls(container);
container.append("div")
.attr("class", "raw-value")
.text("Value: ")
.append("span");

.attr("class", "hash-value")
.text("Hash Value: ")
.append("span");

bitVectorGraphic = blogChart.bitVectorVisual(container,
keepHashBits,

indexLabel: "Register Index: ", valueLabel: "Register Value: " });

container.append("div")
.attr("class", "register-values-label")
.text("Register Values: ");
container.append("div")
.attr("class", "m-label")
.text("m")
.append("span")
.attr("class", "m-value")
.text(" = " + m);
registerValuesGraphic = blogChart.registerValuesVisual(container, registerValues);

container.append("div")
.attr("class", "actual-cardinality")
.text("Actual Cardinality: ")

cardinalityTableGraphic = blogChart.cardinalityTable(container, tableCardinalities);
errorChart = lineChart(container, "% Error");
registerValueHistogram = blogChart.histogram(container,
"register-value-distriubtion", 250/*width*/, 200/*height*/,
series, 0/*starts at zero*/, m/*max count*/,
"Distribution of Register Values",
distributions, means);

reset()/*get to a consistent state*/;
}

function redraw() {
var step = steps[steps.length - 1]/*there's always at least one*/;

d3.select(".raw-value span")
.text(numberFormat(step.value));
d3.select(".hash-value span")
.text(numberFormat(step.hashValue));

d3.select(".actual-cardinality span")
.text(numberFormat(step.cardinality));

bitVectorGraphic.update(step.hashValue, step.registerIndex, step.registerValue);
registerValuesGraphic.update(step);

cardinalityTableGraphic.update(step);

errorChart.redraw();
blogMath.computeTheoreticalRegisterValueDistribution(m, step.cardinality, theoreticalRegisterValueDistribution);
registerValueHistogram.redraw(blogMath.computeMeans(registerValues));
}

// ------------------------------------------------------------------------
function reset() {
stop();
lastTime = 0;

// wait at least the delay to ensure that anything running has stopped
// before redrawing
d3.timer(function(elapsed) {
if(inputAutoScale.property("checked")) inputStepSize.property("value", 1/*reset*/);
valueSet = new Object()/*reset*/;
blogUtil.fillArray(registerValues, m, 0/*reset*/);
blogMath.generateDistribution(registerValues, registerValueDistribution)/*default state*/;
steps.length = 1/*clear (but keep 0th value)*/;

redraw();

return true;
}, delay * 2);
}
function start() {
startStopButton.text("Stop");

stopped = false/*running*/;
lastTime = 0/*reset*/;
d3.timer(run, 0/*start immediately*/);
}
function run(elapsed) {
if(stopped) return stopped;
if((elapsed - lastTime) < delay) return stopped;
lastTime = elapsed;
step();
return stopped;
}
function stop() {
stopped = true;
startStopButton.text("Play");
}
function step() {
var value/*v*/;
var hashValue/*h(v)*/;
var registerIndex/*j*/;
var registerValue/*rho(w)*/;

// if auto-scale then set the scale based on the cardinality
var lastStep = steps[steps.length - 1]/*always at least one step exists by contract*/;
var stepSize = 1/*default*/;
if(inputAutoScale.property("checked")) {
stepSize = Math.min(1000, Math.floor(Math.pow(10, (Math.log(lastStep.cardinality/15/*speed factor*/ + 1/*ensure no 0s*/) / Math.log(10)))));
inputStepSize.property("value", stepSize);
} else { /*don't auto scale*/
stepSize = +inputStepSize.property("value");
if(isNaN(stepSize)) stepSize = 1;
stepSize = Math.max(1/*for sanity*/, stepSize);
}
delay = +inputDelay.property("value");
if(isNaN(delay)) delay = 1/*minimum*/;
delay = Math.max(1/*for sanity*/, delay);
if((lastStep.cardinality + stepSize) > maxCardinality) { stop(); return/*sanity upper-bound*/; }

for(var i=0; i<stepSize; i++) {
value = Math.floor(Math.random() * maxValue) + 1/*disallow '0'*/;
hashValue = blogUtil.jenkinsHash(value);
valueSet["" + value] = value;

registerIndex = hashValue & registerIndexMask;
registerValue = rho(hashValue >>> b);
registerValues[registerIndex] = Math.max(registerValues[registerIndex], registerValue);
}
blogMath.generateDistribution(registerValues, registerValueDistribution);

var cardinality = Object.keys(valueSet).length/*the size of the value set*/;
var estimatedCardinality = blogMath.computeCardinality(m, registerValues);
steps.push({
index: steps.length,
cardinality: cardinality,
value: value,
hashValue: hashValue,
registerIndex: registerIndex,
registerValue: registerValue,

// NOTE: Math.floor() is used to get the estimates to be a whole
// number which is more palatable as a cardinality. By doing
// it here it ensures that all dependent visualizations use
// the same values.
llEstimatedCardinality: Math.floor(estimatedCardinality.llCardinality),
llError: (Math.floor(estimatedCardinality.llCardinality) - cardinality) / cardinality * 100/*%*/,

hllEstimatedCardinality: Math.floor(estimatedCardinality.hllCardinality),
hllCardinalityAlgorithm: estimatedCardinality.hllAlgorithm,
hllError: (Math.floor(estimatedCardinality.hllCardinality) - cardinality) / cardinality * 100/*%*/,

expectationEstimatedCardinality: Math.floor(estimatedCardinality.expectationCardinality),
expectationError: (Math.floor(estimatedCardinality.expectationCardinality) - cardinality) / cardinality * 100/*%*/,
});

redraw();
}

// line chart
function lineChart(parent, yAxisLabel) {
var chart = {}/*'this'*/;

var margin = { top: 20, right: 10, bottom: 0, left: 60 },
width = 400 - margin.left - margin.right,
height = 240 - margin.top - margin.bottom;
var x = d3.scale.linear()
.domain([0, 1/*default*/])
.range([0, width]);
var y = d3.scale.linear()
.domain([-hllErrorPercent * 1.2, hllErrorPercent * 1.2])
.range([height, 0]);
var xAxis = d3.svg.axis()
.scale(x)
.orient("bottom")
.tickFormat(siFormat);
var yAxis = d3.svg.axis()
.scale(y)
.orient("left")
.tickFormat(siFormat);
var llLine = d3.svg.line()
.x(function(d,i) { return x(d.cardinality); })
.y(function(d) { return y(d.llError); });
var hllLine = d3.svg.line()
.x(function(d,i) { return x(d.cardinality); })
.y(function(d) { return y(d.hllError); });

var svg = parent.append("div")
.attr("class", "line-chart")
.append("svg")
.datum(steps)
.attr("width", width + margin.left + margin.right)
.attr("height", height + margin.top + margin.bottom)
.append("g")
.attr("class", "line-chart-data")
.attr("transform", "translate(" + margin.left + "," + margin.top + ")");
createLegend(svg);

svg.append("defs").append("clipPath")
.attr("id", "lineChartClipPath")
.append("rect")
.attr("width", width)
.attr("height", height);

var xAxisG = svg.append("g")
.attr("class", "x axis")
.attr("transform", "translate(0," + height/2 + ")")
.call(xAxis);
var yAxisG = svg.append("g")
.attr("class", "y axis")
.call(yAxis)
yAxisG.append("text")
.text(yAxisLabel)
.attr("transform", "translate(-40, 100) rotate(-90, 0, 0)");
svg.append("rect")/*error bounds*/
.attr("class", "error-rect")
.attr("x", 0)
.attr("y", y(hllErrorPercent))
.attr("width", width)
.attr("height", y(-hllErrorPercent) - y(hllErrorPercent));
svg.append("rect")/*algorithm switch*/
.attr("class", "algorithm-switch-rect")
.attr("visibility", "hidden"/*initially hidden*/)
.attr("y", 0)
.attr("width", 1)
.attr("height", height);

var pathG = svg.append("g")
.attr("clip-path", "url(#lineChartClipPath)");
pathG.append("path")
.attr("class", "line ll")
.attr("d", llLine);
pathG.append("path")
.attr("class", "line ball-bin")
.attr("d", hllLine);
pathG.append("path")
.attr("class", "line hll")
.attr("d", hllLine);

pathG.append("circle")
.attr("class", "point")
.attr("visibility", "hidden")
.attr("r", 3);
pathG.append("text")
.attr("class", "value")
.attr("visibility", "hidden");

chart.redraw = function() {
var lastStep = steps[steps.length - 1];
if(steps.length > 1/*(there is a 0th default value)*/) {
x.domain([0, (lastStep.cardinality * 1.2/*right padding*/)]);
// NOTE: 0 must be in the range so that the x-axis is shown
var yExtent = d3.extent(steps, function(d) { return d.hllError; });
y.domain([(yExtent[0] > -hllErrorPercent ? -hllErrorPercent : yExtent[0]) * 1.2,
(yExtent[1] < hllErrorPercent ? hllErrorPercent : yExtent[1]) * 1.2]);
xAxisG
.attr("transform", "translate(0," + y(0) + ")")
.call(xAxis);
yAxisG.call(yAxis);
d3.selectAll("path.ll")
.datum(steps)
.attr("visibility", "visible")
.attr("d", llLine);
d3.selectAll("path.ball-bin")
.datum(steps.filter(function(d) { return (d.hllCardinalityAlgorithm == ALGO_BALL_BIN); }))
.attr("visibility", "visible")
.attr("d", hllLine);
if(lastStep.hllCardinalityAlgorithm == ALGO_HLL) {
// TODO: get previous step for first element so that there's
// not a gap in the graph when it switches algo
d3.selectAll("path.hll")
.datum(steps.filter(function(d) { return (d.hllCardinalityAlgorithm == ALGO_HLL); }))
.attr("visibility", "visible")
.attr("d", hllLine);
} /* else -- the algorithm didn't switch to HLL yet */

var xPos = x(lastStep.cardinality);
var yPos = y(lastStep.hllError);
d3.select("rect.error-rect")
.attr("y", y(hllErrorPercent))
.attr("height", y(-hllErrorPercent) - y(hllErrorPercent));
d3.select("circle.point")
.attr("visibility", "visible")
.attr("cx", xPos)
.attr("cy", yPos);
d3.select("text.value")
.attr("visibility", "visible")
.attr("transform", "translate(" + (xPos + 4) + "," + (yPos + 5) + ")")
.text(percentFormat(lastStep.hllError) + "%");
} else {/*there are no steps*/
d3.select("rect.algorithm-switch-rect").attr("visibility", "hidden");
d3.selectAll("path.line").attr("visibility", "hidden");
d3.select("circle.point").attr("visibility", "hidden");
d3.select("text.value").attr("visibility", "hidden");
}
};

function createLegend(parent) {
var titleG = parent.append("g")
.attr("class", "title")
.attr("transform", "translate(50, -10)");

var xOffset = 0;
algorithms.forEach(function(d) {
titleG.append("line")
.attr("class", d.className)
.attr("x1", xOffset).attr("y1", -4)
.attr("x2", (xOffset + 10)).attr("y2", -4);
xOffset += 15;
titleG.append("text")
.text(d.name)
.attr("x", xOffset);
xOffset += 50;
});
}

return chart;
}

<h3>Usage</h3>
<ul>
<li><em>Step</em>: Clicking the step button will generate a new random value and insert it into the DV sketch;</li>
<li><em>Play</em>: Clicking the play button will start the simulation running. Random values will be generated and inserted into the sketch with a delay of 'n' ms between each value. The play button will change to the "Stop" button;</li>
<li><em>Reset</em>: Clicking the reset button will stop a running simulation and reset the values to their defaults;</li>
<li><em>Auto Scale</em>: If checked then the step size (the number of values that are generated for each step) will be automatically changed to based on the number of values that have already been added to the sketch. This provides for a smoother visualization across the span of values;</li>
<li><em>Step Size</em>: The number of values that will be generated and inserted into the sketch for each step;</li>
<li><em>Delay</em>: The number of miliseconds between generating new values used when the simulation is played. This can be reduced to make the simulation play faster or increased to make it play slower;</li>
</ul>

<h3>Simulation Elements</h3>
<ul>
<li><em>Value</em>: The last randomly generated value (between 1 and 16,777,215) that was inserted into the sketch. At most 65,335 values will be generated;</li>
<li><em>Hash Value</em>: The <a href="http://en.wikipedia.org/wiki/Jenkins_hash_function">Jenkins</a> hash of the value;</li>
<li><em>Bit String</em>: The binary representation of the lower 24bits of the hash value;</li>
<li><em>Register Index</em>: The lowest <em>b</em> bits used to determine the index of the register whose value is to be updated. (<em>m=2<sup>b</sup></em>)</li>
<li><em>Register Value</em>: The next <em>2<sup>r</sup> - 1</em> bits (where <em>r</em> is the register width) from which the register value -- the run of <em>0</em>s -- is determined. Please note that the register value is the number of leading zeros <em>+1</em> as per the HLL paper;</li>
<li><em>Register Values</em>: All <em>m=64</em> register values. Index 0 is in the upper-left and index 63 is in the lower-right. The register that is to be updated is highlighted in red. Please note that the value is only updated if the value from the bit string is <em>greater</em> than the current value;</li>
<li><em>m</em>: The number of buckets or register values (<em>m=2<sup>b</sup></em>)</li>
<li><em>Actual Cardinality</em>: The cardinality of the set of all inserted values (i.e. the count of distinct values);</li>
<li><em>Algorithm</em>: The distinct value (DV) algorithm;</li>
<li><em>Estimated Cardinality</em>: The cardinality of the set estimated by the algorithm;</li>
<li><em>% Error</em>: The difference between the estimated and actual cardinality over the actual cardinality;</li>
<li><em>Line Graph</em>: Plots the actual cardinality on the x-axis and the % Error on the y-axis for the each algorithm. The initial ball-and-bins estimation of HLL is explicitly shown. The blue area is the theoretical HLL error bounds (<em>1.04 / sqrt(m)</em>);</li>
<li><em>Histogram</em>: The distribution of the theoretical and observed register values along with the arithmetic, geometric and harmonic means of the observed distribution. Please note that the harmonic mean used in HLL is 1/2<sup>M</sup> whereas this is simply 1/M and is present for illustrative purposes only;</li>
</ul>

</ul>
